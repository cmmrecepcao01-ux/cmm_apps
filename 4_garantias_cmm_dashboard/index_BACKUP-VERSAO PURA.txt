// INTELIGÊNCIA RESTAURADA: Mantido o agrupamento original da versão .txt
function categorizeDefect(text) {
    if (!text) return "N/D";
    const t = text.toLowerCase();
    
    // 1. O FILTRO DE CASOS CRÍTICOS (O MAIS GRAVE)
    // Palavras que indicam paralisação grave, quebra de bloco, perda total de força, etc.
    if (t.includes("fundid") || t.includes("calço hidráulico") || t.includes("biela") ||
        t.includes("pane geral") || t.includes("imobilizado") || 
        (t.includes("potência") && t.includes("reduzida")) ||
        (t.includes("perdendo") && t.includes("força")) ||
        (t.includes("falha repentina") && t.includes("ruído"))) {
        return "🚨 CASOS CRÍTICOS"; // Mudamos de "Atenção Especial" para "Casos Críticos"
    }

    // 2. O FILTRO DE RECORRÊNCIA (A CHATEAÇÃO)
    // Identifica se a viatura está voltando pelo mesmo problema repetidas vezes
    if (t.includes("vezes") && (t.includes("encaminhada") || t.includes("retorno") || t.includes("concessionária"))) {
        return "RECORRENTE";
    }
    if (t.includes("recorrente") || t.includes("novamente") && t.includes("mesmo problema")) {
         return "RECORRENTE";
    }

    // 3. AS CATEGORIAS COMUNS (MANTIDAS EXATAMENTE COMO ANTES)
    if (t.includes("revisão") || t.includes("periodic")) return "REVISÃO";
    if (t.includes("coxim") || t.includes("câmbio") || t.includes("embreagem")) return "COXIM/CÂMBIO/EMBREAGEM";
    if (t.includes("longarina") || t.includes("chassi") || t.includes("trinca") || t.includes("fratura")) return "ESTRUTURAL";
    if (t.includes("bateria") || t.includes("sonoro") || t.includes("luminoso") || t.includes("injeção") || t.includes("sirene")) return "ELÉTRICO/SONORO";
    if (t.includes("óleo") || t.includes("motor") || t.includes("filtro") || t.includes("vazamento")) return "MOTOR/ÓLEO";
    
    return "OUTROS";
}

function findVal(row, partial) {
    const key = Object.keys(row).find(k => k.toLowerCase().includes(partial.toLowerCase()));
    return key ? row[key] : "";
}

// CÁLCULO DE DIAS (BLINDADO): Tenta ler a coluna nova, se falhar, calcula manualmente. Nunca fica 0.
function getDias(row) {
    let diasPlanilha = findVal(row, "Dias Baixado");
    if (diasPlanilha !== "" && !isNaN(parseInt(diasPlanilha))) {
        return parseInt(diasPlanilha);
    }
    
    // Fallback: Se a coluna M falhar ou não estiver no cache, calcula pela data
    const dtBaixaStr = findVal(row, MAP.baixa);
    const dtEnvioStr = findVal(row, MAP.envio);
    const resolvido = String(findVal(row, MAP.situacao)).toLowerCase() === 'true';
    if (!dtBaixaStr) return 0;

    const parts = dtBaixaStr.split('/');
    if (parts.length < 3) return 0;
    const dataBaixa = new Date(parts[2], parts[1]-1, parts[0]);
    let dataFim = new Date();

    if (resolvido && dtEnvioStr) {
        const pFim = dtEnvioStr.split(' ')[0].split('/');
        if (pFim.length >= 3) dataFim = new Date(pFim[2], pFim[1]-1, pFim[0]);
    }
    const diff = Math.floor((dataFim - dataBaixa) / (1000 * 60 * 60 * 24));
    return diff > 0 ? diff : 0;
}

function executarBuscaUniversal() {
    const termo = document.getElementById('inputBuscaUniversal').value.trim().toLowerCase();
    const divRes = document.getElementById('resultadoBuscaUniversal');
    
    if(!termo) {
        divRes.innerHTML = '<div style="color:#ff8c00; text-align:center;">Por favor, digite algo para buscar.</div>';
        return;
    }
    
    divRes.innerHTML = '<div style="color:#00aaff; margin-top:20px; text-align:center;">Baixando dados atualizados (sem cache)... ⏳</div>';

    // O "nocache" força o navegador a pegar a versão exata de AGORA da sua planilha, ignorando o atraso de 5 min do Google.
    const URL_FRESQUINHA = URL_BANCO_DADOS + "&nocache=" + new Date().getTime();

    Papa.parse(URL_FRESQUINHA, {
        download: true,
        header: true,
        skipEmptyLines: true,
        complete: function(results) {
            cacheBancoDados = results.data; // Atualiza a memória
            processarResultadosBusca(termo, cacheBancoDados);
        },
        error: function() {
            divRes.innerHTML = '<div style="color:#ff4d6d; text-align:center;">Erro ao acessar a planilha Banco de Dados.</div>';
        }
    });
}

function processarResultadosBusca(termo, dados) {
    const divRes = document.getElementById('resultadoBuscaUniversal');
    
    const encontrados = dados.filter(row => {
        return Object.values(row).some(val => String(val).toLowerCase().includes(termo));
    });

    if (encontrados.length === 0) {
        divRes.innerHTML = `<div style="color:#ff4d6d;text-align:center;margin-top:20px;">Nenhum veículo encontrado com o termo: <b>${termo.toUpperCase()}</b></div>`;
        return;
    }

    let html = `<div style="color:#00ff90;margin-bottom:15px;font-size:0.85rem;font-weight:bold;">Encontrado(s) ${encontrados.length} registro(s).</div>`;
    
    encontrados.forEach((row) => {
        html += `<div style="background:#222;border-left:4px solid #ff3cac;border-radius:6px;padding:15px;margin-bottom:15px;text-align:left;">`;
        
        Object.keys(row).forEach(coluna => {
            // Ignora apenas lixo de colunas extras sem nome geradas pelo Excel
            if (!coluna.includes("Unnamed") && coluna.trim() !== "") {
                
                // TRAVA VISUAL: Se a célula estiver vazia na planilha, exibe o "-"
                let valor = row[coluna];
                let valorDaCelula = (valor !== undefined && valor !== null && String(valor).trim() !== "") ? String(valor).trim() : "-";
                // Se for a coluna do SEI e o número estiver cadastrado no nosso banco de links, vira um link clicável
                if (coluna.toUpperCase().includes("SEI") && MAPA_LINKS_SEI[valorDaCelula]) {
                    valorDaCelula = `<a href="${MAPA_LINKS_SEI[valorDaCelula]}" target="_blank" style="color: #00aaff; font-weight: bold; text-decoration: underline;">${valorDaCelula} 🔗 (Abrir SEI)</a>`;
                }
                
                html += `
                    <div style="margin-bottom:8px; border-bottom:1px solid #333; padding-bottom:4px;">
                        <span style="color:#ff3cac; font-size:0.7rem; font-weight:bold; text-transform:uppercase; display:block;">${coluna}</span>
                        <span style="color:#fff; font-size:0.95rem;">${valorDaCelula}</span>
                    </div>
                `;
            }
        });
        html += `</div>`;
    });
    
    divRes.innerHTML = html;
}

let sortDirection = {}; 
let dadosDaMarcaAtual = []; // <-- CAIXA FORTE: Guarda os dados da marca aberta

function sortTable(coluna, tabelaId) {
    const key = tabelaId + '-' + coluna;
    
    // Inverte a direção (ascendente/descendente)
    if (sortDirection[key] === 'asc') sortDirection[key] = 'desc';
    else sortDirection[key] = 'asc';
    const direction = sortDirection[key];
    
    // Limpa setinhas visuais do cabeçalho
    document.querySelectorAll(`th span[id^="sort-${tabelaId}-"]`).forEach(span => span.innerHTML = '');
    document.getElementById(`sort-${tabelaId}-${coluna}`).innerHTML = direction === 'asc' ? ' ▲' : ' ▼';

    let dadosParaOrdenar = [];
    
    if (tabelaId === 'main') {
        const termoBusca = document.getElementById("searchInput").value.toLowerCase();
        if (termoBusca) {
            dadosParaOrdenar = allData.filter(r => Object.values(r).some(v => String(v).toLowerCase().includes(termoBusca)));
        } else {
            dadosParaOrdenar = [...allData];
        }
    } else {
        // Se for na subpágina, usa a Caixa Forte com os dados corretos!
        dadosParaOrdenar = [...dadosDaMarcaAtual]; 
    }

    // Lógica matemática da ordenação
    dadosParaOrdenar.sort((a, b) => {
        let valA, valB;
        switch (coluna) {
            case 'dias':
                valA = getDias(a); valB = getDias(b); break;
            case 'situacao':
                valA = String(findVal(a, MAP.situacao)).toLowerCase() === 'true' ? 1 : 0;
                valB = String(findVal(b, MAP.situacao)).toLowerCase() === 'true' ? 1 : 0; break;
            case 'data':
                const dA = findVal(a, MAP.data).split(' ')[0].split('/');
                const dB = findVal(b, MAP.data).split(' ')[0].split('/');
                valA = new Date(dA[2], dA[1] - 1, dA[0]).getTime() || 0;
                valB = new Date(dB[2], dB[1] - 1, dB[0]).getTime() || 0; break;
            default: 
                valA = String(findVal(a, MAP[coluna]) || '').toLowerCase();
                valB = String(findVal(b, MAP[coluna]) || '').toLowerCase();
        }
        if (valA < valB) return direction === 'asc' ? -1 : 1;
        if (valA > valB) return direction === 'asc' ? 1 : -1;
        return 0;
    });

    // Renderiza a tabela correta
    if (tabelaId === 'main') {
        renderTable(dadosParaOrdenar);
    } else if (tabelaId === 'brand') {
        renderBrandTable(dadosParaOrdenar);
    }
}

let brandChartInstance = null; // Guarda o gráfico para podermos apagá-lo ao trocar de marca

function renderBrandChart(dados) {
    const ctx = document.getElementById('brandMonthlyChart').getContext('2d');
    
    // Se já existir um gráfico na tela, destruímos para desenhar o novo
    if (brandChartInstance) brandChartInstance.destroy();

    // 1. Matemática: Agrupar as VTRs pelo Mês/Ano da Baixa
    let meses = {};

    dados.forEach(row => {
        let dataStr = findVal(row, MAP.data); // Pega a data (Ex: 29/04/2026)
        if (!dataStr) return;

        let partes = dataStr.split(' ')[0].split('/'); // Separa dia, mês e ano
        if (partes.length >= 3) {
            let mesAno = partes[1] + '/' + partes[2]; // Fica: "04/2026"

            if (!meses[mesAno]) {
                meses[mesAno] = { resolvidos: 0, pendentes: 0 };
            }

            // Conta se está resolvido ou pendente
            const sit = String(findVal(row, MAP.situacao)).toLowerCase() === 'true';
            if (sit) meses[mesAno].resolvidos++;
            else meses[mesAno].pendentes++;
        }
    });

    // 2. Ordenar os meses cronologicamente (Jan, Fev, Mar...)
    let labels = Object.keys(meses).sort((a, b) => {
        let [mA, yA] = a.split('/');
        let [mB, yB] = b.split('/');
        return new Date(yA, mA - 1) - new Date(yB, mB - 1);
    });

    // Separar os números para o gráfico
    let dataResolvidos = labels.map(l => meses[l].resolvidos);
    let dataPendentes = labels.map(l => meses[l].pendentes);

    // 3. Desenhar o Gráfico Bonito
    brandChartInstance = new Chart(ctx, {
        type: 'bar',
        data: {
            labels: labels,
            datasets: [
                {
                    label: 'Resolvidos (OK)',
                    data: dataResolvidos,
                    backgroundColor: '#00ff90',
                    borderRadius: 4
                },
                {
                    label: 'Pendentes (Aguardando)',
                    data: dataPendentes,
                    backgroundColor: '#ff4d6d',
                    borderRadius: 4
                }
            ]
        },
        options: {
            responsive: true,
            plugins: {
                legend: { labels: { color: '#fff', font: { family: 'Inter', size: 12 } } },
                tooltip: { mode: 'index', intersect: false }
            },
            scales: {
                x: { 
                    stacked: true, // Empilha as barras
                    ticks: { color: '#aaa' }, 
                    grid: { color: '#333', display: false } 
                },
                y: { 
                    stacked: true, // Empilha as barras
                    ticks: { color: '#aaa', stepSize: 1 }, 
                    grid: { color: '#333' } 
                }
            }
        }
    });
}

function init() {
    Papa.parse(CSV_URL, {
        download: true, header: true, skipEmptyLines: true,
        complete(results) {
            // Guarda TUDO no histórico
            rawHistoryData = results.data.filter(r => findVal(r, MAP.placa).length > 3);
            
            // Cria um dicionário para agrupar por placa
            let grouped = {};
            rawHistoryData.forEach(row => {
                let p = findVal(row, MAP.placa);
                if (!grouped[p]) grouped[p] = [];
                grouped[p].push(row);
            });
            
            // allData agora será apenas o ÚLTIMO lançamento de cada placa
            allData = Object.keys(grouped).map(p => {
                let history = grouped[p];
                return history[history.length - 1]; // Pega sempre o mais recente
            });

            document.getElementById("loader").style.display = "none";
            document.getElementById("dashboard").style.display = "block";
            renderKPIs(allData);
            renderCharts(allData);
            renderTable(allData);
        }
    });
}
